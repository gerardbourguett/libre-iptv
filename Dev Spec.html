DEPENDENCIAS

Módulos Python

Módulos recomendados por categoría. Todos disponibles vía pip.

python-mpv
Binding de MPV para Python. Ideal para reproducción de streams HLS/MPEG-TS con soporte completo de hardware acceleration.
pip install python-mpv
m3u8
Parser de archivos M3U8. Maneja listas HLS y playlists IPTV. Para M3U simples se recomienda parser propio (más rápido).
pip install m3u8
requests + httpx
HTTP para descargar listas M3U remotas, conectar con Xtream Codes API y obtener datos EPG. httpx para requests async.
pip install requests httpx
lxml / xmltodict
Parseo de guías EPG en formato XMLTV (.xml.gz). lxml es más rápido para archivos grandes; xmltodict más simple para guías pequeñas.
pip install lxml xmltodict
PyQt6
Framework UI recomendado para desktop. Permite integrar MPV directamente en un widget. Alternativa: PySide6 (misma API, licencia LGPL).
pip install PyQt6
Pillow
Procesamiento de logos de canales (tvg-logo). Descarga, resize y caché de imágenes PNG/JPG. Necesario para la UI de canales.
pip install Pillow
aiohttp
Descarga asíncrona de recursos: logos, EPG, verificación de streams. Evita bloquear el hilo de UI durante cargas.
pip install aiohttp
appdirs + platformdirs
Rutas de configuración y caché cross-platform. Ubica correctamente los archivos de configuración en Windows/Mac/Linux.
pip install platformdirs
CORE MODULE

M3U Parser

El parser debe soportar los tres formatos de entrada: archivo local, URL remota y Xtream Codes API. Estructura recomendada:

Clase base

from dataclasses import dataclass, field
from typing import List, Optional
import re, requests

@dataclass
class Channel:
    # Campos del #EXTINF
    name: str
    url: str
    group: str = ""           # group-title
    tvg_id: str = ""          # tvg-id  (para EPG)
    tvg_name: str = ""        # tvg-name
    tvg_logo: str = ""        # tvg-logo URL
    num: int = 0               # tvg-chno
    is_live: bool = True
    is_vod: bool = False
    is_series: bool = False

@dataclass
class Playlist:
    source: str                  # URL o path
    channels: List[Channel] = field(default_factory=list)
    epg_url: str = ""
    groups: List[str] = field(default_factory=list)

class M3UParser:
    EXTINF_RE = re.compile(
        r'#EXTINF:(?P<dur>-?\d+)'
        r'(?:.*?tvg-id="(?P<tvg_id>[^"]*)")?'
        r'(?:.*?tvg-name="(?P<tvg_name>[^"]*)")?'
        r'(?:.*?tvg-logo="(?P<tvg_logo>[^"]*)")?'
        r'(?:.*?tvg-chno="(?P<num>\d+)")?'
        r'(?:.*?group-title="(?P<group>[^"]*)")?'
        r'.*?,(?P<name>.+)'
    )

    def parse(self, source: str) -> Playlist:
        content = self._fetch(source)
        return self._parse_content(content, source)

    def _fetch(self, source: str) -> str:
        if source.startswith("http"):
            r = requests.get(source, timeout=30)
            r.raise_for_status()
            return r.text
        with open(source, encoding="utf-8", errors="replace") as f:
            return f.read()

    def _parse_content(self, content: str, source: str) -> Playlist:
        playlist = Playlist(source=source)
        lines = content.splitlines()
        i, channel_num = 0, 1

        # Extract EPG URL from header
        if lines and lines[0].startswith("#EXTM3U"):
            epg_match = re.search(r'url-tvg="([^"]+)"', lines[0])
            if epg_match:
                playlist.epg_url = epg_match.group(1)
            i = 1

        while i < len(lines):
            line = lines[i].strip()
            if line.startswith("#EXTINF"):
                m = self.EXTINF_RE.match(line)
                if m and i + 1 < len(lines):
                    url = lines[i + 1].strip()
                    if url and not url.startswith("#"):
                        ch = Channel(
                            name=m.group("name").strip(),
                            url=url,
                            group=m.group("group") or "General",
                            tvg_id=m.group("tvg_id") or "",
                            tvg_name=m.group("tvg_name") or "",
                            tvg_logo=m.group("tvg_logo") or "",
                            num=int(m.group("num") or channel_num),
                        )
                        # Detect VOD/Series by group name
                        grp_lower = ch.group.lower()
                        ch.is_vod = any(k in grp_lower for k in ["movie", "pelicula", "film", "vod"])
                        ch.is_series = any(k in grp_lower for k in ["series", "serie", "shows"])
                        ch.is_live = not ch.is_vod and not ch.is_series
                        playlist.channels.append(ch)
                        channel_num += 1
                        i += 2
                        continue
            i += 1

        playlist.groups = list(dict.fromkeys(c.group for c in playlist.channels))
        return playlist

Xtream Codes

class XtreamClient:
    def __init__(self, server: str, username: str, password: str):
        self.base = server.rstrip("/")
        self.auth = {"username": username, "password": password}

    def get_m3u_url(self) -> str:
        # Returns full M3U playlist URL with auth token
        params = "&".join(f"{k}={v}" for k, v in self.auth.items())
        return f"{self.base}/get.php?{params}&type=m3u_plus&output=ts"

    def get_epg_url(self) -> str:
        params = "&".join(f"{k}={v}" for k, v in self.auth.items())
        return f"{self.base}/xmltv.php?{params}"

    def get_account_info(self) -> dict:
        r = requests.get(f"{self.base}/player_api.php", params=self.auth)
        return r.json()  # expiry, max_connections, status

EPG ENGINE

Guía de programación (EPG)

Parser XMLTV

import gzip, xml.etree.ElementTree as ET
from datetime import datetime
from typing import Dict, List

@dataclass
class Program:
    channel_id: str
    title: str
    start: datetime
    stop: datetime
    desc: str = ""
    category: str = ""
    icon: str = ""

class EPGParser:
    def load(self, url_or_path: str) -> Dict[str, List[Program]]:
        content = self._fetch(url_or_path)
        root = ET.fromstring(content)
        epg: Dict[str, List[Program]] = {}

        for prog in root.findall("programme"):
            ch_id = prog.get("channel")
            p = Program(
                channel_id=ch_id,
                title=(prog.find("title") or {}).text or "",
                start=datetime.strptime(prog.get("start"), "%Y%m%d%H%M%S %z"),
                stop=datetime.strptime(prog.get("stop"), "%Y%m%d%H%M%S %z"),
                desc=(prog.find("desc") or {}).text or "",
            )
            epg.setdefault(ch_id, []).append(p)
        return epg

    def _fetch(self, source: str) -> bytes:
        if source.startswith("http"):
            r = requests.get(source, timeout=60)
            data = r.content
        else:
            data = open(source, "rb").read()
        return gzip.decompress(data) if source.endswith(".gz") else data
PLAYER

Player Backend (MPV)

import mpv

class IPTVPlayer:
    def __init__(self, wid=None):
        # wid = window ID para embed en PyQt widget
        self.player = mpv.MPV(
            wid=str(wid) if wid else None,
            vo="gpu",             # Hardware accel
            hwdec="auto-safe",   # Decodificación GPU
            cache=True,
            cache_secs=10,
            demuxer_max_bytes="50MiB",
            log_handler=print,
        )

    def play(self, url: str, headers: dict = {}):
        if headers:
            hdr = ",".join(f"{k}: {v}" for k, v in headers.items())
            self.player["http-header-fields"] = hdr
        self.player.play(url)

    def stop(self): self.player.stop()
    def set_volume(self, v: int): self.player.volume = v
    def toggle_pause(self): self.player.pause = not self.player.pause
    def screenshot(self, path): self.player.screenshot(path)

    @property
    def is_playing(self): return not self.player.pause and self.player.time_pos is not None

TOKENS DE AUTENTICACIÓN EN URL

Algunos proveedores inyectan tokens en la URL del stream. El parser los preserva; no modificar la URL del canal al reproducir.

# → Reproducir directamente sin modificar
self.player.play("https://normielista.xyz:443/play/TOKEN/m3u8")
